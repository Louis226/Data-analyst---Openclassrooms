def recherche_outlier(price):
    Q1 = np.quantile(price, 0.25)
    Q3 = np.quantile(price, 0.75)
    IQ = Q3-Q1
    IQ_left = Q1-1.5*IQ
    IQ_right = Q3+1.5*IQ
    i = list(price.index[(price < IQ_left) | (price > IQ_right)])
    val = list(price[i])
    return i, val
